create table responsavel
(
	id serial primary key,
	nome varchar(40),
        senhaDigital varchar(14)
);

create table departamento
(
    id serial primary key ,
    nome varchar(40),
    sigla varchar(6) primary key,
    id_responsavel int references responsavel(id)
);

create table atividade 
(
        id serial primary key,
        nome varchar(60),
        situacao varchar(15),
        id_departamento int references departamento(id)
);

create table processo
(
    id serial primary key,
    nome varchar(40),
    id_responsavel int references responsavel(id),
    dataDeCriacao timestamp,
    id_atividade int references atividade(id),
    dataDeInicio timestamp,
    dataDeFim timestamp,
    duracaoMaximaEmDias int

);
